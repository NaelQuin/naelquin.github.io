# 📊 Persistência de Dados e Controle de Qualidade Humana

Este manual orienta a gravação incremental dos dados estruturados em tabelas do Excel e o desenvolvimento de um painel visual para auditoria de erros.

## Passo 1: Módulo de Gravação no Excel

Crie o ficheiro `gerenciador_excel.py`. Este módulo lê o objeto Pydantic gerado na etapa anterior, valida se a chave primária (Tag/ID) já existe e faz o append da nova linha de forma segura.

```python
import os
from openpyxl import load_workbook, Workbook

def salvar_no_excel(dados_inquerito, caminho_excel="respostas_consolidadas.xlsx"):
    """Grava os dados estruturados garantindo o ID do Inquérito na primeira coluna."""
    dados_dict = dados_inquerito.model_dump()
    cabecalhos = ["id_inquerito", "nome_respondente", "idade", "satisfacao_geral", "comentarios_adicionais", "confianca_extracao"]
    
    # Cria o ficheiro com cabeçalhos caso ele não exista
    if not os.path.exists(caminho_excel):
        wb = Workbook()
        ws = wb.active
        ws.title = "Inquéritos Consolidados"
        ws.append(cabecalhos)
        wb.save(caminho_excel)
    
    wb = load_workbook(caminho_excel)
    ws = wb.active
    
    # Validação de Duplicidade na Coluna A (Chave Primária)
    for row in ws.iter_rows(min_row=2, max_col=1, values_only=True):
        if row[0] == dados_dict["id_inquerito"]:
            print(f"⚠️ Aviso: O ID {dados_dict['id_inquerito']} já existe na tabela. Ignorando.")
            wb.close()
            return False
            
    # Insere os dados seguindo a ordem correta das colunas
    ws.append([dados_dict[coluna] for coluna in cabecalhos])
    wb.save(caminho_excel)
    wb.close()
    print(f"✅ Dados da Tag {dados_dict['id_inquerito']} gravados com sucesso!")
    return True

```
## Passo 2: Rota de Auditoria Humana com Streamlit
Para cumprir os critérios da **US5**, os estagiários devem fornecer um painel visual que capture inquéritos onde o Gemini indicou baixa confiança (< 0.75), permitindo a correção manual dos dados direto na planilha Excel.
Crie o ficheiro app_auditoria.py:
```python
import streamlit as st
import pandas as pd
from openpyxl import load_workbook
import os

st.set_page_config(page_title="Painel de Auditoria", layout="wide")
st.title("🔍 Painel de Controle de Qualidade (OCR + Gemini)")

CAMINHO_EXCEL = "respostas_consolidadas.xlsx"

if not os.path.exists(CAMINHO_EXCEL):
    st.info("Aguardando o processamento de dados. Ficheiro Excel não encontrado.")
else:
    df = pd.read_excel(CAMINHO_EXCEL)
    
    # Filtra apenas os registos com nota de confiança considerada crítica
    casos_criticos = df[df["confianca_extracao"] < 0.75]
    
    col1, col2 = st.columns(2)
    col1.metric(label="Total de Inquéritos Processados", value=len(df))
    col2.metric(label="Casos para Revisão Humana", value=len(casos_criticos))
    
    st.markdown("---")
    
    if len(casos_criticos) == 0:
        st.success("🎉 Excelente! Todos os inquéritos possuem alta confiança de extração.")
    else:
        st.warning("Atenção: Os seguintes inquéritos requerem validação manual:")
        
        id_selecionado = st.selectbox("Selecione o ID do documento para auditar:", casos_criticos["id_inquerito"].unique())
        linha_atual = df[df["id_inquerito"] == id_selecionado].iloc[0]
        
        # Formulário para edição e correção dos dados
        with st.form("form_correcao"):
            novo_nome = st.text_input("Nome do Respondente", value=linha_atual["nome_respondente"])
            nova_idade = st.number_input("Idade", value=int(linha_atual["idade"]), step=1)
            nova_satisfacao = st.selectbox("Satisfação Geral", ["Satisfeito", "Neutro", "Insatisfeito"], index=["Satisfeito", "Neutro", "Insatisfeito"].index(linha_atual["satisfacao_geral"]))
            novos_comentarios = st.text_area("Comentários Transcritos", value=linha_atual["comentarios_adicionais"])
            
            if st.form_submit_button("Confirmar e Salvar Alterações no Excel"):
                wb = load_workbook(CAMINHO_EXCEL)
                ws = wb.active
                
                # Procura a linha correspondente no Excel para atualizar
                for row in ws.iter_rows(min_row=2):
                    if row[0].value == id_selecionado:
                        row[1].value = novo_nome
                        row[2].value = nova_idade
                        row[3].value = nova_satisfacao
                        row[4].value = novos_comentarios
                        row[5].value = 1.0  # Eleva a confiança para 100% pois foi auditada por um humano
                        break
                        
                wb.save(CAMINHO_EXCEL)
                wb.close()
                st.success(f"💾 Ficheiro updated com sucesso para o ID {id_selecionado}! Recarregue a página.")

```
Para colocar o painel visual em execução no navegador, os alunos devem rodar no terminal:
```bash
streamlit run app_auditoria.py

```

### 📷 Passo 4: Validação e Renomeação Automática de Imagens (Tarefa 1.3)
Após digitalizarem os inquéritos no scanner (em resolução mínima de **300 DPI**, salvos em formato .png ou .jpg), os estagiários podem se deparar com arquivos nomeados aleatoriamente pelo software do scanner (ex: scan_001.png, photo_2.jpg).
Abaixo está o script em Python que os estagiários devem utilizar para validar se as imagens estão na resolução correta e renomeá-las em lote usando uma lista correspondente de IDs físicos que eles etiquetaram.
Crie um arquivo chamado validar_e_renomear.py:
```python
import os
import cv2
from PIL import Image

def verificar_e_renomear_lote(pasta_origem, lista_tags):
    """
    Verifica os requisitos mínimos da imagem digitalizada e renomeia o arquivo 
    para o ID correspondente da etiqueta física.
    """
    # Lista e ordena os arquivos de imagem na pasta
    arquivos = sorted([f for f in os.listdir(pasta_origem) if f.lower().endswith(('.png', '.jpg', '.jpeg'))])
    
    if len(arquivos) != len(lista_tags):
        print(f"❌ Erro de correspondência! Encontradas {len(arquivos)} imagens, mas fornecidas {len(lista_tags)} tags.")
        return

    print("🔍 Iniciando validação e renomeação...")
    
    for arquivo, nova_tag in zip(arquivos, lista_tags):
        caminho_antigo = os.path.join(pasta_origem, arquivo)
        
        # 1. Validar a resolução usando OpenCV
        img = cv2.imread(caminho_antigo)
        altura, largura, _ = img.shape
        
        # Validação básica: Um documento A4 escaneado a 300 DPI possui aprox. 2480x3508 pixels
        if largura < 2000 or altura < 2000:
            print(f"⚠️ Aviso: A imagem {arquivo} pode estar abaixo de 300 DPI ({largura}x{altura}). Verifique a qualidade.")
        
        # 2. Definir a extensão e o novo caminho
        extensao = os.path.splitext(arquivo)[1].lower()
        novo_nome = f"{nova_tag}{extensao}"
        caminho_novo = os.path.join(pasta_origem, novo_nome)
        
        # 3. Renomear o arquivo com a Tag Universal
        os.rename(caminho_antigo, caminho_novo)
        print(f"✅ {arquivo} -> {novo_nome} | Dimensões: {largura}x{altura}px")

if __name__ == "__main__":
    # Simulação da pasta de scans do projeto
    PASTA_IMAGENS = "./scans_temporarios"
    
    # Criando pasta de teste caso não exista
    if not os.path.exists(PASTA_IMAGENS):
        os.makedirs(PASTA_IMAGENS)
        print(f"Pasta '{PASTA_IMAGENS}' criada. Insira algumas imagens de teste nela antes de rodar o script.")
    else:
        # Exemplo de mapeamento sequencial gerado a partir dos autocolantes colados
        # Em cenários reais, essa lista pode ser lida de um arquivo CSV básico.
        tags_físicas = ["INQ-2026-0001", "INQ-2026-0002", "INQ-2026-0003"]
        
        verificar_e_renomear_lote(PASTA_IMAGENS, tags_físicas)

```
#### Para executar o script:
```bash

```
